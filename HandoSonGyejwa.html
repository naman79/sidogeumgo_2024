def cvsToDic(file_path, param_list):
 dict_list = []   
 with open(file_path, 'r', encoding='UTF-8') as f:
    
  for line in f:  
   dicdata = {}
   data = line.replace('\n','').split('\t')
   #print(data)  
   edata = enumerate(data)   
   for i, ele in edata:
    dicdata[param_list[i]] = ele
   dict_list.append(dicdata)  
    
 f.close()
 return dict_list
